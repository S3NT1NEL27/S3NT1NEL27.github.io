    return hashlib.md5(text.encode('utf-8')).hexdigest()

documents = []
metadatas = []
ids = []

print("Reading text files and creating embeddings...")
for filename in tqdm(os.listdir(ARCHIVE_DIR)):
    filepath = os.path.join(ARCHIVE_DIR, filename)
    
    if not os.path.isfile(filepath):
        continue
    
    # Skip non-text files (add more extensions if needed)

def clean_text(text):
    return re.sub(r'\s+', ' ', text).strip()

def generate_id(text, seq):
    return hashlib.md5(f"{text}{seq}".encode()).hexdigest()

def estimate_timestamp(line, line_number, total_lines):
    """Linear interpolation of time across file"""
    if total_lines <= 1:
        return "unknown"
    progress = line_number / total_lines
    # Assume archive spans ~6 months: Oct 2024 → Apr 2025
    start = datetime(2024, 10, 1)
    end = datetime(2025, 4, 1)
    delta = end - start
    est = start + delta * progress
    return est.strftime("%Y-%m-%d")

# ========================= CHUNKING =========================
def chronological_chunk_with_overlap(text, chunk_size=400, overlap=100):
    words = text.split()
    chunks = []
    i = 0
    seq = 0
    while i < len(words):
        chunk_words = words[i:i + chunk_size]
        chunk = " ".join(chunk_words)
        chunks.append((seq, chunk, i, i + len(chunk_words)))
        seq += 1
        i += chunk_size - overlap
    return chunks

# ========================= MAIN INGESTION =========================
print(f"Reading {ARCHIVE_FILE}...")
with open(ARCHIVE_FILE, 'r', encoding='utf-8') as f:
    full_text = f.read()

total_lines = len(full_text.split('\n'))
print(f"File loaded: {len(full_text):,} chars, ~{len(full_text.split()):,} words")

# Split into chunks
print("Chunking with overlap (preserving chronology)...")
chunks = chronological_chunk_with_overlap(full_text, CHUNK_SIZE, OVERLAP)

documents = []
metadatas = []
ids = []

print(f"Embedding and indexing {len(chunks)} chunks...")
for seq, chunk, start_idx, end_idx in tqdm(chunks):
    if len(chunk.strip()) < 50:
        continue  # skip tiny chunks

    # Estimate position in file
    mid_char = (start_idx + end_idx) // 2
    approx_line = int((mid_char / len(full_text)) * total_lines)
    est_time = estimate_timestamp(chunk, approx_line, total_lines)

        "chunk_seq": seq,
        "char_start": start_idx,
        "char_end": end_idx,
        "estimated_date": est_time,
        "word_count": len(chunk.split()),
        "is_overlap": seq > 0 and start_idx < chunks[seq-1][2]  # mark overlap
    })
    ids.append(doc_id)

import chromadb
from chromadb.utils import embedding_functions

def search(query, n=5, date_filter=None, seq_range=None):
    filters = {}
    if date_filter:
        filters["estimated_date"] = {"$gte": date_filter}
    if seq_range:
        filters["chunk_seq"] = {"$gte": seq_range[0], "$lte": seq_range[1]}

    results = collection.query(
        query_texts=[query],
        n_results=n * 3,  # overfetch
        where=filters
    )

    # Sort by sequence to rebuild timeline
    items = zip(results['documents'][0], results['metadatas'][0], results['distances'][0])
    sorted_items = sorted(items, key=lambda x: x[1]['chunk_seq'])

    print(f"\n=== CHRONOLOGICAL CONTEXT FOR: '{query}' ===\n")
    for doc, meta, dist in sorted_items[:n]:
        print(f"[{meta['estimated_date']} | Seq:{meta['chunk_seq']}]")
        print(f"{doc[:500]}{'...' if len(doc)>500 else ''}\n")
        print(f"  (distance: {dist:.4f})\n---")

OVERLAP = 100

print("Loading model...")

embedding_function = embedding_functions.SentenceTransformerEmbeddingFunction(model_name=EMBEDDING_MODEL)

client = chromadb.PersistentClient(path=DB_PATH)
collection = client.get_or_create_collection(name=COLLECTION_NAME, embedding_function=embedding_function)

def clean_text(text):
    return re.sub(r'\s+', ' ', text).strip()

def generate_id(text, seq):
    return hashlib.md5(f"{text}{seq}".encode()).hexdigest()

def chunk_text(text, chunk_size=400, overlap=100):
    words = text.split()
    chunks = []
    i = 0
    seq = 0
    while i < len(words):
        chunk = " ".join(words[i:i + chunk_size])
        chunks.append((seq, chunk, i))
        seq += 1
        i += chunk_size - overlap
    return chunks

print(f"Reading {ARCHIVE_FILE}...")
with open(ARCHIVE_FILE, 'r', encoding='utf-8') as f:
    full_text = f.read()

print(f"Chunking {len(full_text):,} chars...")
chunks = chunk_text(full_text, CHUNK_SIZE, OVERLAP)

documents = []
metadatas = []
ids = []

for seq, chunk, start in tqdm(chunks):
    if len(chunk.strip()) < 50:
        continue
    doc_id = generate_id(chunk, seq)
    documents.append(clean_text(chunk))
    metadatas.append({"chunk_seq": seq, "char_start": start})
    ids.append(doc_id)

print(f"Adding {len(documents)} chunks to DB...")
collection.add(documents=documents, metadatas=metadatas, ids=ids)

print("INGESTION COMPLETE!")
print(f"DB saved at: {DB_PATH}")
print(f"Total vectors: {collection.count()}")
